            \onslide<1-> {

                    \begin{blockcolor}{mDarkTeal}{Problem}

                    \textbf{Instance 1}
                    \begin{itemize}[nosep]
                        \item[$\bullet$] $\lfunc(\pv) = \tfrac{1}{2}\norm{\obs - \dic\pv}{2}^2$
                        \item[$\bullet$] $\pfunc(\pv) = \tfrac{\regtwo}{2}\norm{\pv}{2}^2 + \icvx(\norm{\pv}{\infty} \leq \bigM)$
                        \item[$\bullet$] \texttt{riboflavin} dataset with $\dic \in \kR^{71\times4088}$
                    \end{itemize}
                };
            }

            %
